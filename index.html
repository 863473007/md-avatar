md-avatar

Install

bower install md-avatar

Code

var main = document.getElementById('main');
var content = [
    'Scarletsky',
    '你好',
    'Hello',
    'こんにちは',
    'Bonjour',
    'й',
    '안녕하세요'
];

content.forEach(function(t) {
    var avatar = new mdAvatar({
        size: 100,
        text: t
    });

    main.appendChild(avatar);
});
